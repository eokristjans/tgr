# Falli f skilar f(theta, p1, p2, p3, L1, L2, L3, gamma, x1, x2, y2)
# fyrir gefin gildi  theta, p1, p2, p3, L1, L2, L3, gamma, x1, x2 og y2
def f(theta, p1, p2, p3, L1, L2, L3, gamma, x1, x2, y2):
    A2      = L3*np.cos(theta)-x1
    B2      = L3*np.sin(theta)
    A3      = L2*np.cos(theta+gamma)-x2
    B3      = L2*np.sin(theta+gamma)-y2
    aNum    = p2**2-p1**2-A2**2-B2**2
    bNum    = p3**2-p1**2-A3**2-B3**2
    D       = 2*(A2*B3-B2*A3)
    N1      = B3*aNum-B2*bNum
    N2      =-A3*aNum+A2*bNum 
    return N1**2+N2**2-p1**2*D**2

def fPlot(thetaL, thetaH, fileName, graphTitle, 
          minGraphVal, maxGraphVal, interval):
        # Setjum bili  theta sem vi viljum
        # teikna f(theta) fyrir
        theRange = np.arange(thetaL, thetaH, 0.1)
        # Finnum hmarks og lgmarks gildi f  v bili
        maxFunVal = max(f(theRange, p1, p2, p3, 
                          L1, L2, L3, gamma, x1, x2, y2))
        minFunVal = min(f(theRange, p1, p2, p3, 
                          L1, L2, L3, gamma, x1, x2, y2))
        # Upphafsstillum myndina og sana
        figure, axis = plt.subplots(figsize=(12, 6))
        # Teiknum grafi
        axis.plot(theRange, f(theRange, p1, p2, p3, 
                              L1, L2, L3, gamma, x1, x2, y2))
        # Setjum titil  grafi og sana
        axis.set(xlabel=r"${\Theta}$ [radians]", 
                ylabel="$f({\Theta})$",
                title = graphTitle)
        # Setjum hsta og lgsta gildi  y-s
        axis.set_ylim([minGraphVal,maxGraphVal])
        axis.grid()
        # Vistum myndina
        fileString = fileName + ".png"
        figure.savefig(fileString)
        
        # Setjum minnsta snilega bili  x - snum
        # og y - snum
        plt.xticks(np.arange(thetaL, thetaH, 0.5))
        plt.yticks(np.arange(minGraphVal,maxGraphVal,interval))
        plt.grid(True)
        plt.show()

def fSolver(f, thetaGuess): 
        func = lambda theta : f(theta, p1, p2, p3, 
                                L1, L2, L3, gamma, x1, x2, y2)
        thetaSol = scOpt.fsolve(func, thetaGuess)
        return thetaSol

# Teiknar Stewart Platform
def plotStewartPlatform(x, y, x1, x2, y2, xL2, yL2, xL3, yL3, 
                        plotTitle="Vantar titil!", plotName="Vantar"):
        # Teiknar rihyrning. Byrjum lengst til 
        # vinstri og forum rettsaelis
        # Her er (x, y) = (1,2), 
        #        (xL2, yL2) = (2,3)
        #        (xL3, yL3) = (2,1)
        (triangleX, triangleY)  = [x, xL2, xL3, x],[y, yL2, yL3, y]
        fig, plotObject = plt.subplots()
        plotObject.plot(triangleX, triangleY, 'b', linewidth=2.5)

        # Baetum vid fyrsta akkerinu
        # sem hefur hnit fra (0,0) i (x,y)
        firstAnchorX, firstAnchorY  = [0,x],[0,y]
        plotObject.plot(firstAnchorX, firstAnchorY, 'b')

        # Baetum vid odru akkerinu
        # sem hefur hnit fra (x2, y2)
        # til (xL2, yL2) 
        secondAnchorX, secondAnchorY  = [x2,xL2],[y2,yL2]
        plotObject.plot(secondAnchorX, secondAnchorY, 'b')

        # Baetum vid thridja akkerinu
        # sem hefur hnit fra (xL3, yL3)
        # til (x1, 0)
        thirdAnchorX, thirdAnchorY  = [x1,xL3],[0,yL3]
        plotObject.plot(thirdAnchorX, thirdAnchorY, 'b')

        # Baetum vid punktum fyrir 
        # akkaeri eitt, tvo og thrju 
        # sem hafa hnitin
        # (0,0), (x2, y2) og (x1, 0)
        anchorPointsX, anchorPointsY  = [0, x1, x2], [0, 0, y2]
        plotObject.plot(anchorPointsX, anchorPointsY, 'bo')

        # Baetum vid punktum fyrir
        # hvert horn thrihyrningsins
        plotObject.plot(triangleX, triangleY, 'bo')

        plotObject.set(xlabel="x", 
                ylabel="y",
                title = plotTitle)

        # Vistum myndina
        plotSaveFile = plotName + ".png"
        fig.savefig(plotSaveFile)
        plt.show()

def xyCalc(theta, p1, p2, p3, L1, L2, L3, gamma, x1, x2, y2):
    A2      = L3*np.cos(theta)-x1
    B2      = L3*np.sin(theta)
    A3      = L2*np.cos(theta+gamma)-x2
    B3      = L2*np.sin(theta+gamma)-y2
    aNum    = p2**2-p1**2-A2**2-B2**2
    bNum    = p3**2-p1**2-A3**2-B3**2
    D       = 2*(A2*B3-B2*A3)
    N1      = B3*aNum-B2*bNum
    N2      =-A3*aNum+A2*bNum 
    x = N1/D
    y = N2/D
    return [x, y]

def countRoots(f, p2): 
        deltaTheta = 0.001
        thetaOld = -np.pi
        oldSign = f(thetaOld, p1, p2, p3, 
                    L1, L2, L3, gamma, x1, x2, y2) > 0
        count = 0
        for theta in np.arange(-np.pi, np.pi+deltaTheta, deltaTheta):
                thetaNew = theta
                newSign = f(thetaNew, p1, p2, p3, 
                            L1, L2, L3, gamma, x1, x2, y2) > 0
                if(newSign != oldSign):
                        count += 1
                oldSign = newSign
        return count

Verkefni snst um a hanna sex-leggja svispall. Til eru mismunandi lausnir verkefninu. Ein lausn er annig a hver leggur er tveimur prtum, bir af fastri lengd, annar snertingu vi svispallinn, og hinn hluti hvers leggs snst um sinn eigin snningsxul. nnur lausn er annig a leggirnir geta lengst, en upphafsh eirra er fst.

S hluti verkefnisins sem snr a framvirkri hreyfifri (e. forward kinematics) snst um a finna stasetningu og halla svispallsins egar lengdir leggjanna eru ekktar, svipa v sem er var gert tvva riggja-leggja verkefninu a ofan. a er engin g ekkt jafna til a leysa a verkefni, nema bara a finna nllstvar jfnu svipari og var gert a ofan, .e.a.s. fundin er bestunarlausn me reikniritum. Ltum vigurinn $$\bar{t} = [t_x, t_y, t_z]$$ tkna stasetningu miju svispallsins rvdd, og eru skilgreind rj snningshorn pallsins $\alpha, \beta, \gamma$ um $x, y, z$ sana. Svo eru leggirnir skilgreindir me sex vigrum af fastri lengd, og sex vigrar af breytilegri lengd (v leggirnir eru lengjanlegir) og stasetningarhnit eirra eru $$\bar{b_i} = [b_ix, b_iy, 0]$$ og $$\bar{p_i} = [p_ix, p_iy, 0]$$ fyrir $$i = 1,...,6$$ $z$-hnitin eru ll $0$ v reikningar mia vi a grunnhnit leggjanna su jrinni ea upphafspunkti z-ss. N er hgt a tkna lengdir leggjanna $\bar{l_i} = -\bar{b_i} + \bar{t} + \underline{R}\cdot\bar{p_i}$ ar sem $ \underline{R}$ er snningsfylki, reikna t fr snningshornunum remur.
